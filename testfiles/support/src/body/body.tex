\input{body/colors.tex}

\def\dopage{%
    \null
    \begin{tikzpicture}[remember picture,overlay]
		\ifpaintbg
			\path [fill=color\arabic{page}] (current page.south west) rectangle (current page.north east);
			\node[font=\sffamily\huge,scale=15,text depth=.2pt] at (current page) {\thepage};
		\else
			\node[font=\sffamily\huge,scale=15,text depth=.2pt,color=color\arabic{page}] at (current page) {\thepage};
		\fi
    \end{tikzpicture}%
    \clearpage
}

\ExplSyntaxOn
\newif \ifpaintbg
\cs_if_exist:NF \nobackground { \paintbgtrue }
\int_step_inline:nn { 24 } { \dopage }
\ExplSyntaxOff

\end{document}
